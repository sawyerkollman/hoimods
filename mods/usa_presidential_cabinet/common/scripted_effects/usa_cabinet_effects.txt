###########################################################################
# Presidential Cabinet - USA Overhaul : Scripted Effects
# Helper effects used by the focus tree to drive the cabinet system.
# Everything that GATES content is variable-based, so the tree works even
# if the (isolated) power-balance feature is unavailable on a given patch.
###########################################################################

# Called when the Brain Trust focus is taken. The power balance itself is
# auto-applied to the USA via its definition (default = yes); this just gives
# a little political capital to kick off cabinet-building.
USA_setup_cabinet_balance = {
	add_political_power = 25
}

# Increment the count of filled cabinet seats (gates The Inner Circle focus).
USA_fill_cabinet_seat = {
	add_to_variable = { USA_cabinet_seats_filled = 1 }
}

# A progressive appointment: raises progressive influence and nudges the
# cabinet power balance toward the New Deal / progressive pole (left).
USA_shift_cabinet_progressive = {
	add_to_variable = { USA_progressive_influence = 1 }
	add_power_balance_value = { id = USA_cabinet_balance value = -0.15 }
}

# A conservative appointment: raises conservative influence and nudges the
# cabinet power balance toward the Old Guard pole (right).
USA_shift_cabinet_conservative = {
	add_to_variable = { USA_conservative_influence = 1 }
	add_power_balance_value = { id = USA_cabinet_balance value = 0.15 }
}

###########################################################################
# ERA FLAGS - Set by the election resolution events as each presidential
# term begins. Era-gated focuses and slates read these flags.
#
# CONVENTION (anchored by usa_cabinet.101, the 1936 election outcome):
#   The election held in November of year Y sets USA_era_<Y+4>.
#     1936 -> USA_era_1940    1940 -> USA_era_1944    1944 -> USA_era_1948
#     1948 -> USA_era_1952    1952 -> USA_era_1956
#   USA_era_<Y> therefore means: "the election cycle that precedes the era
#   band beginning in year Y has been resolved (by event, by bypass, or by
#   a dark-twin crisis focus), so era-<Y> content may unlock." Era-gated
#   focuses must still carry their own date floor; the flag alone is set as
#   early as November of Y-4.
#
# Primary setter: the election outcome events (usa_cabinet .101/.103,
# .111-.114, .121-.124, .131-.134, .141-.144).
# Backstop: USA_era_flag_backstop below, run from the monthly pulse in
# usa_cabinet_on_actions.txt, for elections that were bypassed at war, the
# hung 1936 outcome (.102, which sets no era flag), and the
# suspended-election (Caesar) path.
###########################################################################

USA_set_era_flag_1940 = {
	set_country_flag = USA_era_1940
}

USA_set_era_flag_1944 = {
	set_country_flag = USA_era_1944
}

USA_set_era_flag_1948 = {
	set_country_flag = USA_era_1948
}

USA_set_era_flag_1952 = {
	set_country_flag = USA_era_1952
}

USA_set_era_flag_1956 = {
	set_country_flag = USA_era_1956
}

# Date-based era-flag backstop. The election EVENTS are the primary setter;
# this only catches the cases where no outcome event ever fires:
#   - '40/'44 election focuses bypassed because the nation is at war
#     (has_completed_focus is TRUE for bypassed focuses in HOI4, which is
#     exactly what we rely on here);
#   - the hung 1936 outcome (usa_cabinet.102 sets no era flag);
#   - the authoritarian track, where the dark-twin focuses
#     (USA_trunk_third_term_crisis / USA_trunk_suspended_election) replace
#     an election outright, or USA_elections_suspended keeps later ballots
#     from ever being held.
# Each clause is an independent if (not else_if) so a late-start bookmark
# catches up on every era in a single pulse. set_country_flag is idempotent;
# the NOT guards just keep the effect quiet once a flag exists.
USA_era_flag_backstop = {
	if = {
		limit = {
			date > 1936.11.10
			NOT = { has_country_flag = USA_era_1940 }
			OR = {
				has_completed_focus = USA_trunk_election_1936
				has_country_flag = USA_elections_suspended
			}
		}
		USA_set_era_flag_1940 = yes
	}
	if = {
		limit = {
			date > 1940.11.10
			NOT = { has_country_flag = USA_era_1944 }
			OR = {
				has_completed_focus = USA_trunk_election_1940
				has_completed_focus = USA_trunk_third_term_crisis
				has_country_flag = USA_elections_suspended
			}
		}
		USA_set_era_flag_1944 = yes
	}
	if = {
		limit = {
			date > 1944.11.10
			NOT = { has_country_flag = USA_era_1948 }
			OR = {
				has_completed_focus = USA_trunk_election_1944
				has_completed_focus = USA_trunk_suspended_election
				has_country_flag = USA_elections_suspended
			}
		}
		USA_set_era_flag_1948 = yes
	}
	if = {
		limit = {
			date > 1948.11.10
			NOT = { has_country_flag = USA_era_1952 }
			OR = {
				has_completed_focus = USA_trunk_election_1948
				has_country_flag = USA_elections_suspended
			}
		}
		USA_set_era_flag_1952 = yes
	}
	if = {
		limit = {
			date > 1952.11.10
			NOT = { has_country_flag = USA_era_1956 }
			OR = {
				has_completed_focus = USA_trunk_election_1952
				has_country_flag = USA_elections_suspended
			}
		}
		USA_set_era_flag_1956 = yes
	}
}

###########################################################################
# WAR / PEACE STORY FLAGS
###########################################################################

# Monthly fallback for the war-entry / postwar story flags. The on_actions
# (on_declare_war / on_war_relation_added) set USA_at_war the moment war
# touches America; this pulse is the belt to those braces, and is the sole
# setter of USA_postwar.
#   USA_at_war  - "America has been at war" (a permanent historical record;
#                 it is deliberately never cleared).
#   USA_postwar - set once America has been at war and the wars are over.
USA_war_peace_pulse = {
	if = {
		limit = {
			has_war = yes
			NOT = { has_country_flag = USA_at_war }
		}
		set_country_flag = USA_at_war
	}
	else_if = {
		limit = {
			has_war = no
			has_country_flag = USA_at_war
			NOT = { has_country_flag = USA_postwar }
		}
		set_country_flag = USA_postwar
	}
}

###########################################################################
# ELECTION RESOLVER
###########################################################################

# Decide a trunk election that the ballot event has thrown to the country
# at large. Reads the cabinet power balance (progressive pole is NEGATIVE,
# Old Guard pole is POSITIVE) plus the wing commitment flags, and sets
# EXACTLY ONE USA_election_winner_<wing> country flag, clearing the
# previous winner flags first. The ballot events call this and then route
# to the proper outcome event by reading the flag back in a hidden_effect;
# outcome events may also read the flags in option triggers.
# If the power-balance feature is unavailable on a given patch, the
# incumbent (dem) wins by default - the same graceful degradation the
# original inline check in usa_cabinet.100.b used.
USA_resolve_election = {
	clr_country_flag = USA_election_winner_dem
	clr_country_flag = USA_election_winner_gop
	clr_country_flag = USA_election_winner_rad
	clr_country_flag = USA_election_winner_auth
	if = {
		limit = {
			has_country_flag = USA_wing_rad
			has_power_balance = { id = USA_cabinet_balance }
			power_balance_value = { id = USA_cabinet_balance value < -0.3 }
		}
		set_country_flag = USA_election_winner_rad
	}
	else_if = {
		limit = {
			has_country_flag = USA_wing_auth
			has_power_balance = { id = USA_cabinet_balance }
			power_balance_value = { id = USA_cabinet_balance value > 0.3 }
		}
		set_country_flag = USA_election_winner_auth
	}
	else_if = {
		limit = {
			has_power_balance = { id = USA_cabinet_balance }
			power_balance_value = { id = USA_cabinet_balance value > 0.05 }
		}
		set_country_flag = USA_election_winner_gop
	}
	else = {
		set_country_flag = USA_election_winner_dem
	}
}

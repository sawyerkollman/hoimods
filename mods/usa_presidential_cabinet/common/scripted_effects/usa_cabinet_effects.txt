###########################################################################
# Presidential Cabinet - USA Overhaul : Scripted Effects
# Helper effects used by the focus tree to drive the cabinet system.
# Everything that GATES content is variable-based, so the tree works even
# if the (isolated) power-balance feature is unavailable on a given patch.
###########################################################################

# Called when the Brain Trust focus is taken. The power balance itself is
# auto-applied to the USA via its definition (default = yes); this just gives
# a little political capital to kick off cabinet-building.
USA_setup_cabinet_balance = {
	add_political_power = 25
}

# Increment the count of filled cabinet seats (gates The Inner Circle focus).
USA_fill_cabinet_seat = {
	add_to_variable = { USA_cabinet_seats_filled = 1 }
}

# A progressive appointment: raises progressive influence and nudges the
# cabinet power balance toward the New Deal / progressive pole (left).
USA_shift_cabinet_progressive = {
	add_to_variable = { USA_progressive_influence = 1 }
	add_power_balance_value = { id = USA_cabinet_balance value = -0.15 }
}

# A conservative appointment: raises conservative influence and nudges the
# cabinet power balance toward the Old Guard pole (right).
USA_shift_cabinet_conservative = {
	add_to_variable = { USA_conservative_influence = 1 }
	add_power_balance_value = { id = USA_cabinet_balance value = 0.15 }
}

###########################################################################
# ERA FLAGS - Set by the election resolution events as each presidential
# term begins. Era-gated focuses and slates read these flags.
###########################################################################

USA_set_era_flag_1940 = {
	set_country_flag = USA_era_1940
}

USA_set_era_flag_1944 = {
	set_country_flag = USA_era_1944
}

USA_set_era_flag_1948 = {
	set_country_flag = USA_era_1948
}

USA_set_era_flag_1952 = {
	set_country_flag = USA_era_1952
}

# Sweep the cabinet clean between administrations: removes every cabinet
# seat idea from any era and resets the seat counter.
# IMPORTANT: every new cabinet idea id added by later tasks (the 1940-1952
# era slates) MUST be appended to the remove_ideas list below.
USA_clear_cabinet = {
	remove_ideas = {
		USA_cabinet_cordell_hull
		USA_cabinet_isolationist_state
		USA_cabinet_morgenthau
		USA_cabinet_business_treasury
		USA_cabinet_stimson
		USA_cabinet_woodring
		USA_cabinet_murphy
		USA_cabinet_security_ag
	}
	set_variable = { USA_cabinet_seats_filled = 0 }
}

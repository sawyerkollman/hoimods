# THE ELECTION OF 1936 (usa_cabinet.100-104)
# Fired by focus USA_trunk_election_1936. This chain is the pattern every
# later election copies: ballot event -> outcome event -> new-term flavour.
#######################################################################

# The ballot: campaign hard, trust the cabinet's record, or court the fringe
country_event = {
	id = usa_cabinet.100
	title = usa_cabinet.100.t
	desc = usa_cabinet.100.d
	picture = GFX_report_event_usa_congress_speaks

	is_triggered_only = yes

	# Campaign on the New Deal: the historical landslide.
	option = {
		name = usa_cabinet.100.a
		ai_chance = { factor = 70 }
		country_event = { id = usa_cabinet.101 hours = 6 }
	}

	# Let the cabinet's balance decide. USA_resolve_election (scripted
	# effects) weighs the cabinet power balance and the wing flags, then
	# sets exactly one USA_election_winner_* flag, which we read back here.
	# In 1936 no wing commitment flags can exist yet, so only the dem/gop
	# outcomes are reachable; rad/auth route to the hung election for safety.
	option = {
		name = usa_cabinet.100.b
		ai_chance = { factor = 25 }
		USA_resolve_election = yes
		hidden_effect = {
			if = {
				limit = { has_country_flag = USA_election_winner_gop }
				country_event = { id = usa_cabinet.103 hours = 6 }
			}
			else_if = {
				limit = {
					OR = {
						has_country_flag = USA_election_winner_rad
						has_country_flag = USA_election_winner_auth
					}
				}
				country_event = { id = usa_cabinet.102 hours = 6 }
			}
			else = {
				country_event = { id = usa_cabinet.101 hours = 6 }
			}
		}
	}

	# Back a third-party surge. Stand-in gate: the radical anchor focus.
	# TODO(Task 8): tighten to "completed >= 3 radical-wing focuses" once
	# the radical wing exists.
	option = {
		name = usa_cabinet.100.c
		trigger = {
			has_completed_focus = USA_rad_anchor
		}
		ai_chance = { factor = 5 }
		country_event = { id = usa_cabinet.102 hours = 6 }
	}
}

# The Marshall Plan (usa_cabinet.251): fired by USA_dem_marshall_plan.
# The USA offers reconstruction aid to Western European democracies.
# Recipients must exist and be democratic; USA absorbs the cost regardless.
# A fallback option exists if no eligible recipients are present.
country_event = {
	id = usa_cabinet.251
	title = usa_cabinet.251.t
	desc = usa_cabinet.251.d
	picture = GFX_report_event_press_conference

	is_triggered_only = yes

	# Full programme: all willing recipients receive the Marshall aid.
	option = {
		name = usa_cabinet.251.a
		trigger = {
			OR = {
				AND = { country_exists = ENG has_government = democratic }
				AND = { country_exists = FRA has_government = democratic }
				AND = { country_exists = HOL has_government = democratic }
			}
		}
		ai_chance = { factor = 80 }
		add_political_power = -50
		add_ideas = USA_marshall_outlays_spirit
		if = {
			limit = { country_exists = ENG has_government = democratic }
			ENG = {
				add_opinion_modifier = { target = USA modifier = USA_atlantic_charter_partner }
				add_political_power = 25
			}
		}
		if = {
			limit = { country_exists = FRA has_government = democratic }
			FRA = {
				add_opinion_modifier = { target = USA modifier = USA_atlantic_charter_partner }
				add_political_power = 25
			}
		}
		if = {
			limit = { country_exists = HOL has_government = democratic }
			HOL = {
				add_opinion_modifier = { target = USA modifier = USA_atlantic_charter_partner }
				add_political_power = 25
			}
		}
		add_ideas = USA_marshall_plan_spirit
	}

	# Limited programme: some recipients but not all.
	option = {
		name = usa_cabinet.251.b
		trigger = {
			OR = {
				AND = { country_exists = ENG has_government = democratic }
				AND = { country_exists = FRA has_government = democratic }
				AND = { country_exists = HOL has_government = democratic }
			}
		}
		ai_chance = { factor = 15 }
		add_political_power = -25
		if = {
			limit = { country_exists = ENG has_government = democratic }
			ENG = { add_opinion_modifier = { target = USA modifier = USA_atlantic_charter_partner } }
		}
		if = {
			limit = { country_exists = FRA has_government = democratic }
			FRA = { add_opinion_modifier = { target = USA modifier = USA_atlantic_charter_partner } }
		}
		add_ideas = USA_marshall_plan_spirit
	}

	# or non-democratic). The plan goes unspent; still qualifies the focus.
	option = {
		name = usa_cabinet.251.c
		trigger = {
			NOT = { country_exists = ENG has_government = democratic }
			NOT = { country_exists = FRA has_government = democratic }
			NOT = { country_exists = HOL has_government = democratic }
		}
		ai_chance = { factor = 5 }
		add_political_power = 25
	}
}
